00278 /* ------------------- Fine Control over USB Descriptors ------------------- */
00279 /* If you don't want to use the driver's default USB descriptors, you can
00280  * provide our own. These can be provided as (1) fixed length static data in
00281  * flash memory, (2) fixed length static data in RAM or (3) dynamically at
00282  * runtime in the function usbFunctionDescriptor(). See usbdrv.h for more
00283  * information about this function.
00284  * Descriptor handling is configured through the descriptor's properties. If
00285  * no properties are defined or if they are 0, the default descriptor is used.
00286  * Possible properties are:
00287  *   + USB_PROP_IS_DYNAMIC: The data for the descriptor should be fetched
00288  *     at runtime via usbFunctionDescriptor(). If the usbMsgPtr mechanism is
00289  *     used, the data is in FLASH by default. Add property USB_PROP_IS_RAM if
00290  *     you want RAM pointers.
00291  *   + USB_PROP_IS_RAM: The data returned by usbFunctionDescriptor() or found
00292  *     in static memory is in RAM, not in flash memory.
00293  *   + USB_PROP_LENGTH(len): If the data is in static memory (RAM or flash),
00294  *     the driver must know the descriptor's length. The descriptor itself is
00295  *     found at the address of a well known identifier (see below).
00296  * List of static descriptor names (must be declared PROGMEM if in flash):
00297  *   char usbDescriptorDevice[];
00298  *   char usbDescriptorConfiguration[];
00299  *   char usbDescriptorHidReport[];
00300  *   char usbDescriptorString0[];
00301  *   int usbDescriptorStringVendor[];
00302  *   int usbDescriptorStringDevice[];
00303  *   int usbDescriptorStringSerialNumber[];
00304  * Other descriptors can't be provided statically, they must be provided
00305  * dynamically at runtime.
00306  *
00307  * Descriptor properties are or-ed or added together, e.g.:
00308  * #define USB_CFG_DESCR_PROPS_DEVICE   (USB_PROP_IS_RAM | USB_PROP_LENGTH(18))
00309  *
00310  * The following descriptors are defined:
00311  *   USB_CFG_DESCR_PROPS_DEVICE
00312  *   USB_CFG_DESCR_PROPS_CONFIGURATION
00313  *   USB_CFG_DESCR_PROPS_STRINGS
00314  *   USB_CFG_DESCR_PROPS_STRING_0
00315  *   USB_CFG_DESCR_PROPS_STRING_VENDOR
00316  *   USB_CFG_DESCR_PROPS_STRING_PRODUCT
00317  *   USB_CFG_DESCR_PROPS_STRING_SERIAL_NUMBER
00318  *   USB_CFG_DESCR_PROPS_HID
00319  *   USB_CFG_DESCR_PROPS_HID_REPORT
00320  *   USB_CFG_DESCR_PROPS_UNKNOWN (for all descriptors not handled by the driver)
00321  *
00322  * Note about string descriptors: String descriptors are not just strings, they
00323  * are Unicode strings prefixed with a 2 byte header. Example:
00324  * int  serialNumberDescriptor[] = {
00325  *     USB_STRING_DESCRIPTOR_HEADER(6),
00326  *     'S', 'e', 'r', 'i', 'a', 'l'
00327  * };
00328  */
