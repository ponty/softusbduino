00029 ;max stack usage: [ret(2), YL, SREG, YH, [sofError], bitcnt(x5), shift, x1, x2, x3, x4, cnt, ZL, ZH] = 14 bytes
00030 ;nominal frequency: 18 MHz -> 12 cycles per bit
00031 ; Numbers in brackets are clocks counted from center of last sync bit
00032 ; when instruction starts
00033 ;register use in receive loop to receive the data bytes:
00034 ; shift assembles the byte currently being received
00035 ; x1 holds the D+ and D- line state
00036 ; x2 holds the previous line state
00037 ; cnt holds the number of bytes left in the receive buffer
00038 ; x3 holds the higher crc byte (see algorithm below)
00039 ; x4 is used as temporary register for the crc algorithm
00040 ; x5 is used for unstuffing: when unstuffing the last received bit is inverted in shift (to prevent further
00041 ;    unstuffing calls. In the same time the corresponding bit in x5 is cleared to mark the bit as beening iverted
00042 ; zl lower crc value and crc table index
00043 ; zh used for crc table accesses
00044 
00045 ;--------------------------------------------------------------------------------------------------------------
00046 ; CRC mods:
00047 ;  table driven crc checker, Z points to table in prog space
00048 ;   ZL is the lower crc byte, x3 is the higher crc byte
00049 ;   x4 is used as temp register to store different results
00050 ;   the initialization of the crc register is not 0xFFFF but 0xFE54. This is because during the receipt of the
00051 ;   first data byte an virtual zero data byte is added to the crc register, this results in the correct initial
00052 ;   value of 0xFFFF at beginning of the second data byte before the first data byte is added to the crc.
00053 ;   The magic number 0xFE54 results form the crc table: At tabH[0x54] = 0xFF = crcH (required) and
00054 ;   tabL[0x54] = 0x01  ->  crcL = 0x01 xor 0xFE = 0xFF
00055 ;  bitcnt is renamed to x5 and is used for unstuffing purposes, the unstuffing works like in the 12MHz version
00056 ;--------------------------------------------------------------------------------------------------------------
00057 ; CRC algorithm:
00058 ;   The crc register is formed by x3 (higher byte) and ZL (lower byte). The algorithm uses a 'reversed' form
00059 ;   i.e. that it takes the least significant bit first and shifts to the right. So in fact the highest order
00060 ;   bit seen from the polynomial devision point of view is the lsb of ZL. (If this sounds strange to you i
00061 ;   propose a research on CRC :-) )
00062 ;   Each data byte received is xored to ZL, the lower crc byte. This byte now builds the crc
00063 ;   table index. Next the new high byte is loaded from the table and stored in x4 until we have space in x3
00064 ;   (its destination).
00065 ;   Afterwards the lower table is loaded from the table and stored in ZL (the old index is overwritten as
00066 ;   we don't need it anymore. In fact this is a right shift by 8 bits.) Now the old crc high value is xored
00067 ;   to ZL, this is the second shift of the old crc value. Now x4 (the temp reg) is moved to x3 and the crc
00068 ;   calculation is done.
00069 ;   Prior to the first byte the two CRC register have to be initialized to 0xFFFF (as defined in usb spec)
00070 ;   however the crc engine also runs during the receipt of the first byte, therefore x3 and zl are initialized
00071 ;   to a magic number which results in a crc value of 0xFFFF after the first complete byte.
00072 ;
00073 ;   This algorithm is split into the extra cycles of the different bits:
00074 ;   bit7:   XOR the received byte to ZL
00075 ;   bit5:   load the new high byte to x4
00076 ;   bit6:   load the lower xor byte from the table, xor zl and x3, store result in zl (=the new crc low value)
00077 ;           move x4 (the new high byte) to x3, the crc value is ready
00078 ;

00099 macro CRC_CLEANUP_AND_CHECK
00100     ; the last byte has already been xored with the lower crc byte, we have to do the table lookup and xor
00101     ; x3 is the higher crc byte, zl the lower one
00102     ldi     ZH, hi8(usbCrcTableHigh);[+1] get the new high byte from the table
00103     lpm     x2, Z               ;[+2][+3][+4]
00104     ldi     ZH, hi8(usbCrcTableLow);[+5] get the new low xor byte from the table
00105     lpm     ZL, Z               ;[+6][+7][+8]
00106     eor     ZL, x3              ;[+7] xor the old high byte with the value from the table, x2:ZL now holds the crc value
00107     cpi     ZL, 0x01            ;[+8] if the crc is ok we have a fixed remainder value of 0xb001 in x2:ZL (see usb spec)
00108     brne    ignorePacket        ;[+9] detected a crc fault -> paket is ignored and retransmitted by the host
00109     cpi     x2, 0xb0            ;[+10]
00110     brne    ignorePacket        ;[+11] detected a crc fault -> paket is ignored and retransmitted by the host
00111     endm
